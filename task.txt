    function selectBranch() {
      $.post("dirs/dashboard/actions/get_branches.php", {}, function(data) {
        var response = JSON.parse(data);
        if ($.trim(response.isSuccess) === "success") {
          var branches = response.Data;
          $("#filter-branch").empty().append('<option value="">ALL BRANCH</option>');
          branches.forEach(function(branch) {
            $("#filter-branch").append('<option value="' + branch.Branch + '">' + branch.Branch + '</option>');
          });
        } else {
          alert($.trim(response.Data));
        }
      });
    }
